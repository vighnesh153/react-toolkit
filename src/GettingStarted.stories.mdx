# Getting started with [@vighnesh153/react-toolkit](https://www.npmjs.com/package/@vighnesh153/react-toolkit)

<br />

### Installation
```sh
npm install --save @vighnesh153/react-toolkit @material-ui/core --legacy-peer-deps
```

### Usage
```tsx
import React from "react";

import {
  Button,
  ButtonProps,
} from "@vighnesh153/react-toolkit";

const App: React.FC = (props) => {
  return (
    <div>
      <Button
        rvLabel={'Button Text'}
        rvColor={'primary'}
      />
    </div>
  );
};

export default App;
```
